{{- /* Photo Stack shortcode — renders a stack of photos with flip interaction. Usage: {{< photos "img1.jpg|Caption 1,img2.jpg,img3.jpg|Caption 3" >}} Separate src and optional label with | per entry. */ -}} {{- $raw := "" -}} {{- if .IsNamedParams -}} {{- $raw = .Get "src" -}} {{- else -}} {{- $raw = .Get 0 -}} {{- end -}} {{- if not $raw -}} {{- errorf "photos shortcode requires a comma-separated list of image paths" -}} {{- end -}} {{- $entries := split $raw "," -}} {{- $id := printf "photoStack_%d" .Ordinal -}} {{- $data := slice -}} {{- range $entries -}} {{- $entry := trim . " " -}} {{- $parts := split $entry "|" -}} {{- $src := trim (index $parts 0) " " -}} {{- $label := "" -}} {{- if gt (len $parts) 1 -}} {{- $label = trim (index $parts 1) " " -}} {{- end -}} {{- $data = $data | append (dict "src" $src "label" $label) -}} {{- end -}}
