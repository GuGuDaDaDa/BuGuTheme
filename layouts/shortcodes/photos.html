{{- /* Photo Stack shortcode — renders a stack of photos with flip interaction. Usage: {{< photos >}} {{< photo src="img1.jpg" caption="Caption 1" >}} {{< photo src="img2.jpg" >}} {{< /photos >}} */ -}} {{- $id := printf "photoStack_%d" .Ordinal -}} {{- /* Collect JSON objects emitted by each nested {{< photo >}} call */ -}} {{- $inner := .Inner | strings.TrimSpace -}} {{- $items := slice -}} {{- range (split $inner "\n") -}} {{- $line := strings.TrimSpace . -}} {{- if and $line (strings.HasPrefix $line "{") -}} {{- $items = $items | append ($line | transform.Unmarshal) -}} {{- end -}} {{- end -}} {{- if not $items -}} {{- errorf "photos shortcode requires at least one {{< photo >}} entry" -}} {{- end -}}
